def run_quiz(questions):
    score = 0
    for q in questions:
        print("\n" + q["question"])
        for option in q["options"]:
            print(option)
        answer = input("Enter your answer (A/B/C/D): ").strip().upper()
        if answer == q["answer"]:
            print("‚úÖ Correct!")
            score += 1
        else:
            print(f"‚ùå Wrong! The correct answer is {q['answer']}")
    print(f"\nüéâ You got {score} out of {len(questions)} correct.")
